# zkCli.sh
```sh
# 指定ip和端口，zkCli.sh默认进入的是2181端口
# sh zkCli.sh -server localhsot:2170
sh zkCli.sh -server ip:port
```



## zkServer.sh

```sh
# 启动
sh zkServer.sh start
# 查看状态
sh zkServersh status
```
